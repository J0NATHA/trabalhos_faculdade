#include <stdio.h>
#include <stdlib.h>
#include <locale.h>
#define TAM 3

int main()
{
    setlocale(LC_ALL, "");
    int mat[TAM][TAM];
    int i,j, soma=0;

    for(i=0;i<TAM;i++){
        for(j=0;j<TAM;j++){
            printf("\n Valor da posiÃ£o %i, %i -> ",i,j);
            scanf("%i", &mat[i][j]);
            if((i+j)% 2 == 0)
                soma+= mat[i][j];
        }
    }

    printf("Valor da soma -> %i",soma);

    return 0;
}
